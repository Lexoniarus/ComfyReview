{# Shared label badges. Matching and label resolution happen in services. This template only renders. Rules: - One label per category when present. - Modifiers can be multiple labels (limited). {%- endif -%}
{%- if show_character and it.character_name -%} {{ _badge((('character: ' ~ it.character_name) if show_prefix else it.character_name), True, 'character') }} {%- endif -%} {%- if it.scene_name -%} {{ _badge((('scene: ' ~ it.scene_name) if show_prefix else it.scene_name), False, 'scene') }} {%- endif -%} {%- if it.outfit_name -%} {{ _badge((('outfit: ' ~ it.outfit_name) if show_prefix else it.outfit_name), False, 'outfit') }} {%- endif -%} {%- if it.pose_name -%} {{ _badge((('pose: ' ~ it.pose_name) if show_prefix else it.pose_name), False, 'pose') }} {%- endif -%} {%- if it.expression_name -%} {{ _badge((('expression: ' ~ it.expression_name) if show_prefix else it.expression_name), False, 'expression') }} {%- endif -%} {%- if it.light_name -%} {{ _badge((('light: ' ~ it.light_name) if show_prefix else it.light_name), False, 'light') }} {%- endif -%} {%- if it.modifiers and it.modifiers|length > 0 -%} {%- for m in it.modifiers[:modifier_limit] -%} {{ _badge((('modifier: ' ~ m) if show_prefix else m), False, 'modifier') }} {%- endfor -%} {%- if it.modifiers|length > modifier_limit -%} {{ _badge((('modifier: +' ~ (it.modifiers|length - modifier_limit) ~ ' more') if show_prefix else ('+' ~ (it.modifiers|length - modifier_limit) ~ ' more')), False, 'modifier') }} {%- endif -%} {%- endif -%}
